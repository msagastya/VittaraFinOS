Create: `lib/logic/investment_event_type.dart`
```dart
class InvestmentEventType {
  static const String buy = 'buy';
  static const String sell = 'sell';
  static const String sip = 'sip';
  static const String dividend = 'dividend';
  static const String redeem = 'redeem';
  static const String redemption = 'redemption';
  static const String maturity = 'maturity';
  static const String payout = 'payout';
  static const String withdrawal = 'withdrawal';
  static const String decrease = 'decrease';
  static const String increase = 'increase';
  static const String create = 'create';

  static bool isCredit(String? type) {
    return const {dividend, sell, decrease, redeem, redemption, withdrawal, payout, maturity}
        .contains(type);
  }

  static String getDisplayLabel(String? type) {
    switch (type) {
      case buy: case increase: return 'Buy';
      case sell: case decrease: return 'Sell';
      case sip: return 'SIP';
      case dividend: return 'Dividend';
      case redeem: case redemption: return 'Redeem';
      case maturity: return 'Maturity';
      case payout: return 'Payout';
      case withdrawal: return 'Withdrawal';
      case create: return 'Created';
      default: return type != null ? type[0].toUpperCase() + type.substring(1) : 'Activity';
    }
  }
}
```
Then migrate the 5 sites above to use this class instead of inline string checks.
